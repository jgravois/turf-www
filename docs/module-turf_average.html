turf.average

Calculates the average value of a field for points within a set of polygons.

Parameters

Name Type Description
polygons FeatureCollection

a featurecollection of features with Polygon geometries
points FeatureCollection

a featurecollection of features with Point geometries
field string

the field from which to pull values from the properties of each Feature in the points FeatureCollection
outputField string

the field from which to put values from the properties of each Feature in the polygons FeatureCollection

Returns

FeatureCollection

a collection of features with Polygon geometries and the value of outField set to an average.

Example

var poly1 = turf.polygon([[[0,0],[10,0],[10,10], [0,10]]])
var poly2 = turf.polygon([[[10,0],[20,10],[20,20], [20,0]]])
var polyFC = turf.featurecollection([poly1, poly2])
var pt1 = turf.point(5,5, {population: 200})
var pt2 = turf.point(1,3, {population: 600})
var pt3 = turf.point(14,2, {population: 100})
var pt4 = turf.point(13,1, {population: 200})
var pt5 = turf.point(19,7, {population: 300})
var ptFC = turf.featurecollection([pt1, pt2, pt3, pt4, pt5])
var averaged = turf.average(polyFC, ptFC, 'population', 'pop_avg')
console.log(averaged.features[0].properties.pop_avg) // 400
console.log(averaged.features[1].properties.pop_avg) // 200
